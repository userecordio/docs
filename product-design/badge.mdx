## Variants

| Variant | Background | Text | Use case |
|---------|-----------|------|----------|
| `default` | `--base-primary` | `--base-primary-foreground` | General label |
| `secondary` | `--base-secondary` | `--base-secondary-foreground` | Neutral / subtle |
| `destructive` | `--base-destructive` | `--base-destructive-foreground` | Error, danger, removal |
| `outline` | transparent | `--base-foreground` | Minimal, bordered |
| `success` | `--tailwind-colors-green-100` | `--tailwind-colors-green-800` | Positive status |
| `warning` | `--tailwind-colors-amber-100` | `--tailwind-colors-amber-800` | Caution state |

## React / TSX

```tsx
import { cn } from "@/lib/utils";
import { HTMLAttributes } from "react";

type BadgeVariant = "default" | "secondary" | "destructive" | "outline" | "success" | "warning";

interface BadgeProps extends HTMLAttributes<HTMLSpanElement> {
  variant?: BadgeVariant;
}

const variantClass: Record<BadgeVariant, string> = {
  default: "badge--default",
  secondary: "badge--secondary",
  destructive: "badge--destructive",
  outline: "badge--outline",
  success: "badge--success",
  warning: "badge--warning",
};

export function Badge({ variant = "default", className, ...props }: BadgeProps) {
  return (
    <span
      className={cn("badge", variantClass[variant], className)}
      {...props}
    />
  );
}
```

## Usage Examples

```tsx
{/* Status badges */}
<Badge variant="success">Active</Badge>
<Badge variant="warning">Pending</Badge>
<Badge variant="destructive">Failed</Badge>
<Badge variant="secondary">Draft</Badge>

{/* Category tag */}
<Badge variant="outline">Design</Badge>

{/* Count badge */}
<Badge>12 new</Badge>

{/* With icon */}
<Badge variant="success">
  <CheckIcon size={10} />
  Verified
</Badge>
```

## Accessibility

<Note>
  Badges are purely visual. If a badge conveys important status information, also communicate it to screen readers via `aria-label` on the parent element or `<span className="sr-only">` inside the badge.
</Note>

```tsx
<button aria-label="Notifications, 3 unread">
  <BellIcon />
  <Badge aria-hidden="true">3</Badge>
</button>
```
