  "pink", "rose",
];
const steps = [50, 100, 200, 300, 400, 500, 600, 700, 800, 900, 950];

  ["Orange", "--custom-colors-prim-primary---orange"],

      <p className="text-sm font-medium text-muted-foreground">Design Tokens</p>

        Semantic tokens drive the UI. Hover any swatch and click to copy its CSS variable.
      </p>

      <h2 className="mt-12 text-2xl font-semibold tracking-tight">Semantic</h2>
